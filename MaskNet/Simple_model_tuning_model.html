    class_folder = os.path.join(dataset_folder, class_name, 'images')
    mask_folder = os.path.join(dataset_folder, class_name, 'lung masks')

    for filename in os.listdir(class_folder):
        img_path = os.path.join(class_folder, filename)
        mask_path = os.path.join(mask_folder, filename)

        # Load and resize the image

        img_array = image.img_to_array(img)

        # Load and resize the mask

        mask_array = image.img_to_array(mask)

        # Appending to the lists

        masks.append(mask_array)

# Define the model architecture
def create_model(optimizer='adam', dropout_rate=0.2):
    model = Sequential()
    model.add(Conv2D(32, (3, 3), activation='relu', input_shape=(224, 224, 1)))
    model.add(MaxPooling2D((2, 2)))
    model.add(Conv2D(64, (3, 3), activation='relu'))
    model.add(MaxPooling2D((2, 2)))
    model.add(Conv2D(128, (3, 3), activation='relu'))
    model.add(MaxPooling2D((2, 2)))
    model.add(Flatten())
    model.add(Dense(128, activation='relu'))
    model.add(Dropout(dropout_rate))
    model.add(Dense(len(classes), activation='softmax'))
    model.compile(optimizer=optimizer, loss='categorical_crossentropy', metrics=['accuracy'])
    return model

# Splitting the data into training and validation sets
X_train, X_val, y_train, y_val = train_test_split(images, labels, test_size=0.2, random_state=42)

# Defining Hyperparameters for tuning
optimizers = ['adam', 'rmsprop', 'sgd']
dropout_rates = [0.2, 0.3, 0.4]

best_accuracy = 0
best_hyperparameters = {}

# Iteratting over hyperparameters
for optimizer in optimizers:
    for dropout_rate in dropout_rates:
        # Creating  and compiled the model with current hyperparameters
        model = create_model(optimizer=optimizer, dropout_rate=dropout_rate)
        
        # Training the model
        history = model.fit(X_train, y_train, validation_data=(X_val, y_val), epochs=10, batch_size=32, verbose=0)
        
        # Evaluating the model
        loss, accuracy = model.evaluate(X_val, y_val)
        
        # Updating best hyperparameters if current model performs good reults
        if accuracy > best_accuracy:
            best_accuracy = accuracy
            best_hyperparameters = {'optimizer': optimizer, 'dropout_rate': dropout_rate}

print("Best hyperparameters:", best_hyperparameters)
print("Best validation accuracy:", best_accuracy)
