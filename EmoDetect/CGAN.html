import numpy as np
import tensorflow as tf
from tensorflow.keras.preprocessing.image import ImageDataGenerator

# Function to preprocess the images
def preprocess_dataset(image_label_pairs, target_size=(64, 64), batch_size=32, validation_split=0.5):
    # Data augmentation and normalization parameters
    train_datagen = ImageDataGenerator(
        rescale=1. / 255,
        rotation_range=10,
        width_shift_range=0.1,
        height_shift_range=0.1,
        shear_range=0.1,
        zoom_range=0.1,
        horizontal_flip=True
    )

    # Preprocessing images and labels
    images = []
    labels = []
    for image_path, label in image_label_pairs:
        image = tf.keras.preprocessing.image.load_img(image_path, target_size=target_size)
        image = tf.keras.preprocessing.image.img_to_array(image)
        image = train_datagen.random_transform(image)
        image = train_datagen.standardize(image)
        images.append(image)
        labels.append(label)

    # Converting lists to numpy arrays
    images = np.array(images)
    labels = np.array(labels)

    # Split the dataset into training and validation sets
    indices = np.arange(len(images))
    np.random.shuffle(indices)
    split_idx = int(len(images) * (1 - validation_split))
    train_indices = indices[:split_idx]
    val_indices = indices[split_idx:]

    train_images, train_labels = images[train_indices], labels[train_indices]
    val_images, val_labels = images[val_indices], labels[val_indices]

    # Creating generators
    train_generator = tf.keras.preprocessing.image.ImageDataGenerator().flow(train_images, train_labels, batch_size=batch_size)
    val_generator = tf.keras.preprocessing.image.ImageDataGenerator().flow(val_images, val_labels, batch_size=batch_size)

    return train_generator, val_generator

from tensorflow.keras.layers import Input, Dense, Reshape, Flatten, Dropout, Concatenate, BatchNormalization, Activation, LeakyReLU, UpSampling2D, Conv2D, ReLU
from tensorflow.keras.models import Model

# Function to build the generator
def build_generator(latent_dim, num_classes):
    noise = Input(shape=(latent_dim,))
    label = Input(shape=(num_classes,))

    label_embedding = Dense(128 * 16 * 16)(label)
    label_embedding = Reshape((16, 16, 128))(label_embedding)

    # Projecting noise input into a tensor with same shape as label embedding
    noise_embedding = Dense(16 * 16 * 128)(noise)
    noise_embedding = Reshape((16, 16, 128))(noise_embedding)

    # Concatenating noise and label embedding
    concatenated = Concatenate(axis=-1)([noise_embedding, label_embedding])

    # Generator layers
    x = UpSampling2D()(concatenated)
    x = Conv2D(256, kernel_size=3, padding="same")(x)
    x = BatchNormalization()(x)
    x = ReLU()(x)

    x = UpSampling2D()(x)
    x = Conv2D(128, kernel_size=3, padding="same")(x)
    x = BatchNormalization()(x)
    x = ReLU()(x)

    x = Conv2D(64, kernel_size=3, padding="same")(x)
    x = BatchNormalization()(x)
    x = ReLU()(x)

    x = Conv2D(32, kernel_size=3, padding="same")(x)
    x = BatchNormalization()(x)
    x = ReLU()(x)

    x = Conv2D(3, kernel_size=3, padding="same")(x)
    output = Activation("tanh")(x)

    return Model([noise, label], output)

# Function to build the discriminator
def build_discriminator(img_shape):
    model = Sequential()

    model.add(Conv2D(32, kernel_size=3, strides=2, input_shape=img_shape, padding="same"))
    model.add(LeakyReLU(alpha=0.2))
    model.add(Dropout(0.5))

    model.add(Conv2D(64, kernel_size=3, strides=2, padding="same"))
    model.add(ZeroPadding2D(padding=((0,1),(0,1))))
    model.add(BatchNormalization(momentum=0.8))
    model.add(LeakyReLU(alpha=0.2))
    model.add(Dropout(0.5))

    model.add(Conv2D(128, kernel_size=3, strides=2, padding="same"))
    model.add(BatchNormalization(momentum=0.8))
    model.add(LeakyReLU(alpha=0.2))
    model.add(Dropout(0.5))

    model.add(Conv2D(256, kernel_size=3, strides=1, padding="same"))
    model.add(BatchNormalization(momentum=0.8))
    model.add(LeakyReLU(alpha=0.2))
    model.add(Dropout(0.5))

    model.add(Flatten())
    model.add(Dense(1, activation='sigmoid'))

    img = Input(shape=img_shape)
    validity = model(img)

    return Model(img, validity)

import numpy as np
import tensorflow as tf
from tensorflow.keras.utils import to_categorical

# Training the CGAN
def train_cgan(generator, discriminator, combined, train_generator, epochs, batches_per_epoch, latent_dim, num_classes):
    for epoch in range(epochs):
        for i, (imgs, labels) in enumerate(train_generator):
            if i >= batches_per_epoch:
                break

            # Generating noise for the batch
            noise = np.random.normal(0, 1, (imgs.shape[0], latent_dim))

            # Generating fake images using the generator
            gen_imgs = generator.predict([noise, to_categorical(labels, num_classes)])

            # Training discriminator
            d_loss_real = discriminator.train_on_batch(imgs, np.ones((imgs.shape[0], 1)))
            d_loss_fake = discriminator.train_on_batch(gen_imgs, np.zeros((imgs.shape[0], 1)))
            d_loss = 0.5 * np.add(d_loss_real, d_loss_fake)

            # Training generator
            sampled_labels = np.random.randint(0, num_classes, imgs.shape[0])
            sampled_labels = to_categorical(sampled_labels, num_classes)
            g_loss = combined.train_on_batch([noise, sampled_labels], np.ones((imgs.shape[0], 1)))

            # Checking progress
            print(f"Epoch {epoch + 1}/{epochs}, Batch {i + 1}/{batches_per_epoch}, [D loss: {d_loss[0]}, acc.: {100 * d_loss[1]}] [G loss: {g_loss}]")

# Evaluating the CGAN
def evaluate_cgan(generator, discriminator, combined, val_generator, epochs, batches_per_epoch, latent_dim, num_classes):
    # Initializing lists to store losses and accuracies
    d_losses = []
    g_losses = []
    d_accuracies = []

    for epoch in range(epochs):
        for i, (imgs, labels) in enumerate(val_generator):
            if i >= batches_per_epoch:
                break  # Exit the loop if reached the specified number of batches per epoch

            # Generating noise for the batch
            noise = np.random.normal(0, 1, (imgs.shape[0], latent_dim))

            # Generating fake images using the generator
            gen_imgs = generator.predict([noise, to_categorical(labels, num_classes)])

            # Evaluating the discriminator
            d_loss_real = discriminator.evaluate(imgs, np.ones((imgs.shape[0], 1)), verbose=0)
            d_loss_fake = discriminator.evaluate(gen_imgs, np.zeros((imgs.shape[0], 1)), verbose=0)
            d_loss = 0.5 * np.add(d_loss_real, d_loss_fake)
            d_losses.append(d_loss[0])
            d_accuracies.append(d_loss[1])

            # Evaluating the generator
            g_loss = combined.evaluate([noise, to_categorical(labels, num_classes)], np.ones((imgs.shape[0], 1)), verbose=0)
            g_losses.append(g_loss)

            # Checking evaluation progress
            print(f"Epoch {epoch + 1}/{epochs}, Batch {i + 1}/{batches_per_epoch}, [D loss: {d_loss[0]}, acc.: {100 * d_loss[1]}] [G loss: {g_loss}]")

    # Calculating average losses and accuracies
    avg_d_loss = np.mean(d_losses)
    avg_g_loss = np.mean(g_losses)
    avg_d_accuracy = np.mean(d_accuracies)

    # Printing average evaluation results
    print(f"Avg. Discriminator Loss: {avg_d_loss}, Avg. Generator Loss: {avg_g_loss}, Avg. Discriminator Accuracy: {avg_d_accuracy}")

# Function to generate images
def generate_one_image_per_class(generator, num_classes, latent_dim):
    # Initializing an array to store the generated images
    generated_images = []

    # Generating one image per class
    for class_label in range(num_classes):
        noise = np.random.normal(0, 1, (1, latent_dim))

        label = np.zeros((1, num_classes))
        label[0, class_label] = 1

        # Generating image using the generator
        generated_image = generator.predict([noise, label])

        # Appending generated image to the list
        generated_images.append(generated_image)

    # Converting generated image list to a numpy array
    generated_images = np.array(generated_images)

    return generated_images

def calculate_is(images, batch_size=32):
    images_preprocessed = preprocess_images(images)

    # Predicting from the inception model
    preds = inception_model.predict(images_preprocessed, batch_size=batch_size)

    # Computing marginal distribution
    p_y = np.mean(preds, axis=0)

    p_yx = np.exp(preds) / np.sum(np.exp(preds), axis=1, keepdims=True)

    # Calculating Kullback-Leibler divergence for each image
    kl_divs = np.sum(p_yx * (np.log(p_yx + 1e-16) - np.log(p_y + 1e-16)), axis=1)

    # Calculating the Inception Score
    is_score = np.exp(np.mean(kl_divs))
    return is_score

def calculate_fid(real_images, generated_images):
    real_images_preprocessed = preprocess_images(real_images)
    generated_images_preprocessed = preprocess_images(generated_images)

    # Getting features from Inception model
    real_features = inception_model.predict(real_images_preprocessed)
    generated_features = inception_model.predict(generated_images_preprocessed)

    # Calculating mean and covariance of real and generated features
    mu_real = np.mean(real_features, axis=0)
    mu_generated = np.mean(generated_features, axis=0)
    cov_real = np.cov(real_features, rowvar=False)
    cov_generated = np.cov(generated_features, rowvar=False)

    # Computing Frechet distance
    diff = mu_real - mu_generated
    cov_sqrt = sqrtm(cov_real.dot(cov_generated))
    if np.iscomplexobj(cov_sqrt):
        cov_sqrt = cov_sqrt.real
    fid = np.dot(diff, diff) + np.trace(cov_real + cov_generated - 2 * cov_sqrt)

    return fid
