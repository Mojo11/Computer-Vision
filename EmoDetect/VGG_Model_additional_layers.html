import cv2
import numpy as np
from tensorflow.keras.preprocessing.image import img_to_array, load_img

def custom_preprocessing_function(image):
    # Convert the PIL Image to a numpy array in float format for compatibility with cv2
    img_array = img_to_array(image).astype('float32')
    
    # Convert to grayscale
    gray_img = cv2.cvtColor(img_array, cv2.COLOR_RGB2GRAY)
    
    # Apply Gaussian Blur
    blurred_img = cv2.GaussianBlur(gray_img, (5, 5), 0)

    # Apply Histogram Equalization, ensure input is uint8
    equalized_img = cv2.equalizeHist(blurred_img.astype('uint8'))

    # If the original image was RGB, convert back from grayscale to RGB
    if img_array.shape[-1] == 3:
        equalized_img = cv2.cvtColor(equalized_img, cv2.COLOR_GRAY2RGB)

    # Convert the processed image back to float32 to be compatible with ImageDataGenerator rescaling
    equalized_img = equalized_img.astype('float32')

    return equalized_img

from sklearn.metrics import roc_auc_score, confusion_matrix
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

def plot_confusion_matrix(cm, class_names):
    """
    Plots a confusion matrix using Seaborn's heatmap.
    """
    plt.figure(figsize=(10, 7))
    sns.heatmap(cm, annot=True, fmt="d", cmap="Blues", xticklabels=class_names, yticklabels=class_names)
    plt.title('Confusion Matrix')
    plt.ylabel('Actual Labels')
    plt.xlabel('Predicted Labels')
    plt.show()

from tensorflow.keras.applications import VGG16
from tensorflow.keras import layers, models, optimizers, regularizers
from tensorflow.keras.preprocessing.image import ImageDataGenerator
from tensorflow.keras.callbacks import ReduceLROnPlateau, EarlyStopping

# Load VGG16 base model for fine-tuning
base_model = VGG16(weights='imagenet', include_top=False, input_shape=(128, 128, 3))

# Fine-tune the last convolutional block
for layer in base_model.layers[:-4]:
    layer.trainable = False
for layer in base_model.layers[-4:]:
    layer.trainable = True

# Construct the head model with increased regularization
head_model = layers.Flatten(name='flatten')(base_model.output)
head_model = layers.Dense(128, activation='relu', kernel_regularizer=regularizers.l2(1e-4))(head_model)
head_model = layers.Dropout(0.5)(head_model)  
head_model = layers.Dense(5, activation='softmax')(head_model)

# Compile the model with the adjusted learning rate
model = models.Model(inputs=base_model.input, outputs=head_model)
model.compile(optimizer=optimizers.RMSprop(learning_rate=1e-5),  
              loss='categorical_crossentropy',
              metrics=['accuracy'])
